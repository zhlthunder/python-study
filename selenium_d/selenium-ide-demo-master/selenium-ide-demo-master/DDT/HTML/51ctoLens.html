学生学习时长测试案例
store 1 n
store 0 totalLen
open /selenium/51cto-lens-data.html
storeEval window.document.getElementsByTagName('tr').length rows
storeEval new Array(Number(storedVars['rows'])) phrase
while ${n} <= ${rows}
storeText //tr[${n}]/td ph
runScript javascript{storedVars.phrase[storedVars.n] = storedVars.ph}
storeEval ${n} + 1 n
endWhile
store 1 n
while ${n} < ${rows}
echo javascript{storedVars.phrase[storedVars.n]}
open javascript{storedVars.phrase[storedVars.n]}
storeText css=span.red len
storeEval Number(storedVars['totalLen']) + Number(storedVars['len']) totalLen
storeEval ${n} + 1 n
endWhile
echo 总共学习时长：${totalLen}
echo javascript{'平均学习时长：' + storedVars['totalLen']/storedVars['rows']}
