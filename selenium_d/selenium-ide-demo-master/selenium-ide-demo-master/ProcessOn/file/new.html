新建 Flowchart 流程图的各种模板流程图测试案例
store 1 m
open http://sample.wangding.in/selenium/processon-new-file-data.html
storeEval window.document.getElementsByTagName('tr').length rows
storeEval new Array(storedVars.rows) charts
storeEval new Array(storedVars.rows) nums
while ${m} <= ${rows}
storeText //tr[${m}]/td[1] ph1
runScript javascript{storedVars.charts[storedVars.m] = storedVars.ph1}
storeText //tr[${m}]/td[2] ph2
runScript javascript{storedVars.nums[storedVars.m] = storedVars.ph2}
storeEval ${m} + 1 m
endWhile
store 1 m
while ${m} <= ${rows}
echo javascript{storedVars.charts[storedVars.m] + '---' + storedVars.nums[storedVars.m]}
store 1 n
echo javascript{storedVars.nums[storedVars.m]}
echo javascript{storedVars.n <= storedVars.nums[storedVars.m]}
while javascript{storedVars.n <= storedVars.nums[storedVars.m]}
open new
click javascript{storedVars.charts[storedVars.m]}
gotoIf javascript{storedVars.n == 6} CHANGE
goto NORMAL
label CHANGE
click //div[4]/ul/li[2]
storeEval ${n} - 4 n
runScript javascript{storedVars.nums[storedVars.m] -= 4}
label NORMAL
waitForText //div[2]/div[1] 空模板
click //div[2]/div[${n}]
click id=chart_title
storeEval 'flowChart' + Math.round(Math.random()*1000) fileName
type id=chart_title ${fileName}
clickAndWait id=template_select_ok
assertTitle ${fileName}*ProcessOn*
clickAndWait css=span.ico.ico_goback
assertTitle ProcessOn - 我的文件
waitForText css=div.title 标题
verifyTextPresent ${fileName}
storeEval ${n} + 1 n
endWhile
storeEval ${m} + 1 m
endWhile
